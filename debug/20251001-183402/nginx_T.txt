server {
  listen 80;
  server_name _;
  root /var/www/html;
  index index.php index.html;

  # /admin ??/var/www/html/web/admin (alias) + PHP 蹂댁젙
  location ^~ /admin/ {
    alias /var/www/html/web/admin/;
    index index.php index.html;
    try_files $uri $uri/ /admin/index.php?$args;
  }
  location ~ ^/admin/.*\.php$ {
    include        fastcgi_params;
    fastcgi_param  SCRIPT_FILENAME $request_filename;  # alias ?듭떖
    fastcgi_pass   php:9000;
  }

  # ?섎㉧吏 寃쎈줈
  location / {
    try_files $uri $uri/ /index.php?$args;
  }
  location ~ \.php$ {
    include        fastcgi_params;
    fastcgi_param  SCRIPT_FILENAME $document_root$fastcgi_script_name;
    fastcgi_pass   php:9000;
  }
}

# configuration file /etc/nginx/conf.d/site-herocomics.conf:
server {
    listen 80;
    server_name localhost _;

    root /var/www/html;
    index index.php index.html;
    client_max_body_size 512m;

    # --- 1) /admin/*.php ??臾댁“嫄?FPM ?쇰줈 ---
    location ~ ^/admin/(.+)\.php$ {
        default_type text/html;
        add_header X-Handler "admin-php" always;
        include fastcgi_params;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME /var/www/html/web/admin/$1.php;
        fastcgi_param SCRIPT_NAME     /admin/$1.php;
        fastcgi_pass php:9000;
    }

    # --- 2) /admin ?뺤쟻/?몃뜳??---
    location /admin/ {
        add_header X-Handler "admin-static" always;
        root /var/www/html/web;         # => /var/www/html/web/admin/...
        index index.php;
        try_files $uri $uri/ /admin/index.php?$args;
    }

    # --- 3) 猷⑦듃 ?뺤쟻/?몃뜳??---
    location / {
        add_header X-Handler "root-static" always;
        try_files $uri $uri/ /index.php?$args;
    }

    # --- 4) ?쇰컲 PHP (admin ?ㅼ뿉 ?? ---
    location ~ \.php$ {
        add_header X-Handler "generic-php" always;
        try_files $uri =404;
        include fastcgi_params;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_pass php:9000;
    }

    # ?뺤쟻 罹먯떆
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|webp|svg|woff|woff2)$ {
        expires 7d; access_log off;
    }
}
